const { default: makeWASocket, useMultiFileAuthState } = require('@whiskeysockets/baileys'); const pino = require('pino'); const readline = require('readline'); const fs = require('fs'); const path = require('path'); const { myTypingState, subscribedJids, killSwitches } = require('./src/config/state'); const handleMessage = require('./src/handlers/message'); const handleCall = require('./src/handlers/call'); const handlePresence = require('./src/handlers/presence'); // ========================================== // TERMINAL INTERACTION SETUP // ========================================== const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); const question = (text) => new Promise((resolve) => rl.question(text, resolve)); // ========================================== // MAIN BOT INITIALIZATION FUNCTION // ========================================== async function startStatusBot() { const sessionPath = path.join(__dirname, 'session_info'); const credsPath = path.join(sessionPath, 'creds.json'); // ========================================== // DUAL HYBRID LOGIN SYSTEM // ========================================== if (!fs.existsSync(credsPath)) { console.log('\n======================================================'); console.log('[🚀 BABIYA-MD LOGIN SYSTEM]'); console.log('ඔයාගේ Session ID එක හෝ WhatsApp Number එක (947xxxxxxxx) ඇතුළත් කරන්න.'); console.log('======================================================'); let userInput = await question('\n[?] Input here : '); userInput = userInput.trim(); if (!userInput) { console.log('[ERROR] Input එක හිස්ව තියන්න බැහැ! Exiting...'); process.exit(1); } // 1. SESSION ID ක්‍රමය if (userInput.length > 30 || userInput.includes(';;;')) { try { if (!fs.existsSync(sessionPath)) { fs.mkdirSync(sessionPath, { recursive: true }); } let sessionId = userInput; if (sessionId.includes(';;;')) { sessionId = sessionId.split(';;;')[1]; } const decodedData = Buffer.from(sessionId, 'base64').toString('utf-8'); fs.writeFileSync(credsPath, decodedData, 'utf-8'); console.log('\n[INFO] Session ID සාර්ථකව ඇතුළත් කරගත්තා! Connecting...'); } catch (err) { console.error('\n[ERROR] Invalid Session ID format!', err.message); process.exit(1); } } // 2. PHONE NUMBER ක්‍රමය else { let phoneNumber = userInput.replace(/[^0-9]/g, ''); if (phoneNumber) { if (!fs.existsSync(sessionPath)) { fs.mkdirSync(sessionPath, { recursive: true }); } console.log(`\n[INFO] Connecting using phone number: ${phoneNumber}`); global.tempPhoneNumber = phoneNumber; } else { console.log('[ERROR] අවිධිමත් දුරකථන අංකයක්! Exiting...'); process.exit(1); } } } // Setup multi-file authentication state const { state, saveCreds } = await useMultiFileAuthState('session_info'); // Create a new WhatsApp connection socket const sock = makeWASocket({ auth: state, printQRInTerminal: false, logger: pino({ level: "silent" }), browser: ["Ubuntu", "Chrome", "20.0.04"] }); // ========================================== // WHATSAPP PAIRING CODE REGISTRATION // ========================================== if (!sock.authState.creds.me && global.tempPhoneNumber) { setTimeout(async () => { try { const code = await sock.requestPairingCode(global.tempPhoneNumber); console.log(`\n[!] Pairing Code: => ${code}\n`); delete global.tempPhoneNumber; } catch (err) { console.error('[ERROR] Pairing code එක ඉල්ලද්දී අවුලක් වුණා:', err.message); } }, 6000); } // Save credentials when updated sock.ev.on('creds.update', saveCreds); // ========================================== // CONNECTION STATUS UPDATE HANDLER // ========================================== sock.ev.on('connection.update', async (update) => { const { connection, lastDisconnect } = update; if (connection === 'close') { const statusCode = lastDisconnect?.error?.output?.statusCode; // 401 කියන්නේ සෙශන් එක Invalid හෝ Logout වුණු අවස්ථාව const shouldReconnect = statusCode !== 401; myTypingState.clear(); subscribedJids.clear(); for (let timer of killSwitches.values()) clearTimeout(timer); killSwitches.clear(); if (shouldReconnect) { console.log('[INFO] Connection එක වැහුණා. නැවත Connect වෙනවා...'); startStatusBot(); } else { // 🔥 FIXED: සෙශන් එක බ්ලොක් නම් පරණ ෆයිල් මකනවා, එවිට ඊළඟ පාර අලුතින්ම ID එක ඉල්ලනවා console.log('\n[❌ ERROR] Session එකExpired හෝ Invalid! පරණ ෆයිල් මකනවා...'); try { if (fs.existsSync(sessionPath)) { fs.rmSync(sessionPath, { recursive: true, force: true }); } } catch (e) {} console.log('[!] කරුණාකර බොට්ව නැවත Run කර අලුත් Session ID එකක් ලබාදෙන්න.\n'); process.exit(1); } } else if (connection === 'open') { console.log('\n[SUCCESS] Bot Active! Debug logs on.\n'); await sock.sendPresenceUpdate('available'); } }); // ========================================== // ATTACH MODULAR HANDLERS // ========================================== handleMessage(sock); handleCall(sock); handlePresence(sock); } startStatusBot();
